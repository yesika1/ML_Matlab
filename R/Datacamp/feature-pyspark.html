def read_df_inferSchema(dataset):
    ''' Load Spark dataframe with inferred schema
    arg: dataset, it is the name of the file'''
    
    dataset = dataiku.Dataset(dataset)

    location = dataset.get_location_info()
    files = dataset.get_files_info()
    schema = dataset.read_schema()

    directory = location['info']['path']
    global_paths = files['globalPaths']
    hdfs_paths = [g['path'] for g in global_paths]
    full_paths = [ directory + p for p in hdfs_paths]
    col_names = [s['name'] for s in schema]

    df = sqlContext.read.load(full_paths, format = 'com.databricks.spark.csv', inferSchema='true',delimiter='\t').toDF(*col_names)
    return df

# Calcaulate the missing percentages for each variable. 
def missing_values(dataframe):
    '''Display percentage of missing values within each feature
    Arg: spark dataframe
    return: pandas dataframe
    '''
    rows = dataframe.count() # count total rows of the dataframe
    
    # Calculate the total number of missing value for each column variable, then convert to Pandas dataframe, then transpose. 
    mis_val = dataframe.select(*(sum(col(c).isNull().cast("int")).alias(c) for c in dataframe.columns)).toPandas().transpose()
    
    # insert one column on the right side to show the missing percentages
    mis_val.insert(1, 'Percentage of missing value', 100 * mis_val[0] / rows)
    
    # sort the table by the missing percentage descending
    mis_val.sort_values(['Percentage of missing value'], ascending=[False], inplace=True)
    
    # rename the first columns as "count" of missing
    mis_val = mis_val.rename(columns={0:'count'})
    
    # insert one column on the left side to show the labels of each variable.
    mis_val.insert(0, 'Variable label', mis_val.index.values)
    
    # return a Pandas table summarizing the missing percentages for each variable.
    return mis_val

def drop_missing (dataframe, missing_threshold):
    """
    drop columns that have missing percentage >= missing_threshold
    """
    mis_val = missing_values(dataframe)
    df_new = dataframe.select([c for c in dataframe.columns if mis_val.loc[c, 'Percentage of missing value']<missing_threshold])
    drop_var = list(set(dataframe.columns) - set(df_new.columns))
    return {'df_new': df_new,
            'missing_columns': mis_val,
            'drop_var': drop_var}

def var_type (dataframe):
    """
    """
    string_list = []
    timeStamp_list=[]
    numeric_list = []

    for name, dtype in dataframe.dtypes:

            string_list.append(name)
            count+=1
        if dtype.startswith('timestamp'):
            timeStamp_list.append(name)
        if dtype.startswith('int') or dtype.startswith('double') or dtype.startswith('bigint'):
            numeric_list.append(name)
        #else:
            #print("Error! There are columns that are not recognized in dtype:", name)

from pyspark.sql.functions import col, countDistinct

def distinguish_categorical_var (dataframe, count_threshold):
    """
    This function is to distinguish between categorical/discrete data and numerical data;
    Input parameter: a PySpark dataframe and a count threshold (for example, 100) to define categorical data;
    Return: a name list of categorical variables,
            a name list of numerical variables,
            count:the total number of categorical ones,
            and the distinct count: for each column. 
    """
    #rows = dataframe.count()
    categorical_var = []
    numerical_var = []
    count = 0
    # count the distinct counts for each variable in this dataframe
    distinct_count = dataframe.agg(*(countDistinct(col(c)).alias(c) for c in dataframe.columns)).toPandas().transpose()
    distinct_count = distinct_count.rename(columns={0:'distinctCount'})
    # df_categorical = dataframe.select([i for i in dataframe.columns if distinct_count.loc[i, 'distinctCount']<=100])
    for i in dataframe.columns:
        if distinct_count.loc[i, "distinctCount"]<=count_threshold:
            categorical_var.append(i)
            count +=1
        else:
            numerical_var.append(i)
    return {'categorical_list':categorical_var, 
            'continous_list': numerical_var, 
            'categorical_count': count, 
            'distinct_count': distinct_count}

from pyspark.ml.feature import OneHotEncoderEstimator

def transform_OneHotStringCategorical (df, indexed_categorical_list):
    """
    This function is to one-hot encoding after label encoded the String type categorical variables under PySpark.
    Input parameter: a dataframe and a name list of the categorical variables
    Return: transformed/indexed dataframe.
    """
    df_indexed = df.select([x for x in indexed_categorical_list])
    OneHot_col =[]
    count = 0
    columns = indexed_categorical_list
    outputColumns =[]
    for columnname in columns:
        outputColumns.append('OneHotVec_'+columnname) # one hot vector output
        
    onehots = OneHotEncoderEstimator(inputCols=columns, outputCols=outputColumns)
    df_ohindexed = onehots.fit(df).transform(df)
    
    for column in df_ohindexed.columns:
        if column.startswith('OneHotVec_'):
